%{
#include <stdio.h>
#include <stdlib.h>

int yylex();
int yyerror(const char *s);
%}

%token NUMBER

%left '+' '-'
%left '*' '/'

%%

input:
    input line
  | /* empty */
  ;

line:
    expr '\n'   { printf("Result = %d\n", $1); }
  | '\n'
  ;

expr:
      NUMBER
    | expr '+' expr    { $$ = $1 + $3; }
    | expr '-' expr    { $$ = $1 - $3; }
    | expr '*' expr    { $$ = $1 * $3; }
    | expr '/' expr    { 
                          if ($3 == 0) {
                              printf("Error: Division by zero!\n");
                              $$ = 0;
                          } else {
                              $$ = $1 / $3;
                          }
                        }
    | '(' expr ')'      { $$ = $2; }
    ;

%%

int main() {
    printf("Enter expression:\n");
    yyparse();
    return 0;
}

int yyerror(const char *s) {
    printf("Syntax error!\n");
    return 0;
}
