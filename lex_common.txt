/**************** REGULAR EXPRESSIONS ****************/

/* 1. L = {a, b, c} */
^(a|b|c)$          { printf("MATCHED: %s\n", yytext); }

/* 2. L = {dad} */
^dad$              { printf("MATCHED: %s\n", yytext); }

/* 3. L = c* */
^c*$               { printf("MATCHED: %s\n", yytext); }

/* 4. L = specific words */
^(abccb|abcbc|aacbcb|aacbbc)$   { printf("MATCHED: %s\n", yytext); }

/* 5. L = all b's or all c's */
^(b+|c+)$          { printf("MATCHED: %s\n", yytext); }

/* 6. L = even length (a|b)* */
^([ab]{2})*$       { printf("MATCHED: %s\n", yytext); }

/* 7. L = ab b+ cc a{3,} */
^abb+ccaaa+$       { printf("MATCHED: %s\n", yytext); }

/* 8. L contains substring baab */
.*baab.*           { printf("MATCHED: %s\n", yytext); }

/* 9. L contains at least one aa */
.*aa.*             { printf("MATCHED: %s\n", yytext); }

/* 10. Even a's then odd b's */
^(aa)*(b(b)*)$     { if(strlen(yytext)%2==1) printf("MATCHED: %s\n", yytext); }

/* 11. Contains baa AND aaba */
^(?=.*baa)(?=.*aaba).*   { printf("MATCHED: %s\n", yytext); }

/* 12. Odd number of b's */
^[ab]*b([ab]*b[ab]*b)*$  { printf("MATCHED: %s\n", yytext); }

/* 13. Number of a’s divisible by 3 */
^b*(ab*ab*ab*)*$    { printf("MATCHED: %s\n", yytext); }

/* 14. No more than one aa */
^([^a]|a(?!a))*aa?([^a]|a(?!a))*$   { printf("MATCHED: %s\n", yytext); }

/* 15. Contains neither aa nor bb */
^(ab|ba)*a?$        { printf("MATCHED: %s\n", yytext); }

/* 16. Does NOT contain aaa */
^(a(?!aa)|b)*$      { printf("MATCHED: %s\n", yytext); }

/* 17. Exactly one aaa */
^(a(?!aa)|b)*aaa(a(?!aa)|b)*$   { printf("MATCHED: %s\n", yytext); }

/* 18. all aa must appear before any bb */
^(aa)*a*(bb)*b*$    { printf("MATCHED: %s\n", yytext); }

/* 19. Does not contain bab */
^(?!.*bab)[ab]*$    { printf("MATCHED: %s\n", yytext); }

/**************** DEFAULT RULES ****************/

.|\n  { /* ignore other inputs */ }
